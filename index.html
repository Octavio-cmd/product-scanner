function doPost(e){try{var d=JSON.parse(e.postData.contents);var ss=SpreadsheetApp.getActiveSpreadsheet();var sh=ss.getSheetByName('Clothing')||ss.insertSheet('Clothing');if(sh.getLastRow()===0)sh.appendRow(['SKU','Brand','Category','Size','Color','Condition','Defects','Notes','Front','Back','Tag','Detail','Date']);var fld=DriveApp.getFoldersByName('SavvyPhotos').hasNext()?DriveApp.getFoldersByName('SavvyPhotos').next():DriveApp.createFolder('SavvyPhotos');var sub=fld.createFolder(d.sku);var urls={};['front','back','tag','detail'].forEach(function(k){if(d.photos&&d.photos[k]){var b=d.photos[k];var bl=Utilities.newBlob(Utilities.base64Decode(b.split(',')[1]),'image/jpeg',d.sku+'_'+k+'.jpg');var f=sub.createFile(bl);f.setSharing(DriveApp.Access.ANYONE_WITH_LINK,DriveApp.Permission.VIEW);urls[k]=f.getUrl();}});sh.appendRow([d.sku,d.brand,d.category,d.size,d.color,d.condition,(d.defects||[]).join(', '),d.notes||'',urls.front||'',urls.back||'',urls.tag||'',urls.detail||'',new Date().toLocaleString()]);return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);}catch(err){return ContentService.createTextOutput(JSON.stringify({ok:false,err:err.message})).setMimeType(ContentService.MimeType.JSON);}}
